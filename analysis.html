// analysis.js import { extractFeatures } from "./js/feature.js"; import { predict, getGrade } from "./js/model.js"; import { loadModel } from "./js/autosave.js"; // ================= INIT ================= loadModel(); // ================= UI ================= const input = document.getElementById("imageInput"); const label = document.getElementById("fileLabel"); const button = document.getElementById("analyzeBtn"); const loading = document.getElementById("loading"); const results = document.getElementById("results"); const previewContainer = document.getElementById("previewContainer"); const categorySelect = document.getElementById("categorySelect"); // 🔥 DATA STORAGE untuk export let analysisResults = []; // ================= PREVIEW ================= input.addEventListener("change", () => { const files = input.files; label.innerText = `${files.length} files selected`; previewContainer.innerHTML = ""; Array.from(files).forEach(file => { const img = document.createElement("img"); img.src = URL.createObjectURL(file); previewContainer.appendChild(img); }); }); // ================= ANALYZE ================= button.addEventListener("click", async () => { const files = input.files; const category = categorySelect.value; if (!files.length) { alert("Please select images first"); return; } results.innerHTML = ""; analysisResults = []; loading.style.display = "block"; for (let i = 0; i < files.length; i++) { const file = files[i]; try { const feature = await extractFeatures(file); const pred = predict(category, feature); if (!pred) { results.innerHTML += `
${file.name}
❌ Model not trained
`; continue; } const grade = getGrade(pred.percentage); // simpan untuk export analysisResults.push({ file: file.name, prediction: pred.percentage, grade: grade, confidence: Number(pred.confidence.toFixed(2)) }); // tampilkan UI results.innerHTML += `
${file.name}
Prediction: ${pred.percentage}%
Grade: ${grade}
Confidence: ${pred.confidence.toFixed(2)}
`; } catch (err) { results.innerHTML += `
${file.name}
❌ Error processing image
`; } } loading.style.display = "none"; // tampilkan tombol export setelah analisis if (analysisResults.length > 0) { showExportButton(); } }); // ================= EXPORT BUTTON ================= function showExportButton() { const btn = document.createElement("button"); btn.innerText = "Download Excel"; btn.className = "cp-btn cp-btn-primary"; btn.style.marginTop = "15px"; btn.onclick = exportToExcel; results.appendChild(btn); } // ================= EXPORT FUNCTION ================= function exportToExcel() { // format sesuai request const data = analysisResults.map(row => ({ "Nama file": row.file, "Prediction Nutrition partikel (%)": row.prediction, "Grade": row.grade, "confidence rate": row.confidence })); const worksheet = XLSX.utils.json_to_sheet(data); const workbook = XLSX.utils.book_new(); XLSX.utils.book_append_sheet(workbook, worksheet, "Results"); XLSX.writeFile(workbook, "tubulus_analysis.xlsx"); }
